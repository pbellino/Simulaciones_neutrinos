\section{Introducción}

\begin{frame}{ \textbf{$\nu$IOLETA}: \large{\textbf{n}eutrino \textbf{I}nteraction \textbf{O}bservation with a \textbf{L}ow \textbf{E}nergy \textbf{T}hreshold \textbf{A}rray }}
        \begin{itemize}
          \item<+-> Experimento para medir neutrinos producidos en un reactor nuclear
          \item<+-> Propiedades determinadas a partir de interacciones neutrino-electron y neutrino-núcleo
          \item<+-> Es un esfuerzo colaborativo entre varias instituciones y países
          \item<+-> A corto plazo se instalará un experimento en la Central Nuclear Atucha II \item<+-> De particular interés resulta la Dispersión Elástica Coherente neutrino-Núcleo (CE$\nu$NS)
          \item<+-> Se utilizarán sensores CCD con tecnología Skipper
          \item<+-> Estos detectores poseen umbrales de detección de 15 eV (4 e$^-$) y ruido de lectura subelectrónico
          \item<+-> Detección de electrones individuales desde 0 a 100000 e$^-$
        \end{itemize}
        \uncover<+->{
            \vfill
        \begin{center}
            \href{www.violetaexperiment.com}{\fbox{www.violetaexperiment.com}}
        \end{center}
        }
\end{frame}

\section{Simulaciones}

\begin{frame}{Objetivos de las simulaciones}
    \begin{block}{ I) Determinación del factor de quenching}
        \begin{itemize}
            \item Modelar experimentos con Skippers-CCD con el mayor detalle posible
            \item Estimar el factor de quenching al comparar experimentos y simulación
        \end{itemize}
    \end{block}
    \begin{block}{II) Optimización del blindaje del experimento}
        \begin{itemize}
             \item Reducir la radiación de fondo que le llegue al detector
             \item Limitación de tamaño y materiales debido a estar en una central nuclear
             \item Optimización de materiales para blindaje de campos mixtos
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Herramientas de simulación}
        \begin{itemize}
          \item<+-> Será necesario simular partículas con un amplio rango de energías (eV - GeV)
          \item<+-> Una opción es trabajar con distintos programas para distintos rangos
          \item<+-> Para validar los modelos y programas se pueden comparar entre ellos (en los rangos donde solapan el rango de trabajo) y también comparar con datos experimentales
          \item<+-> En esta charla se mostrarán los avances realizados para validar el modelado de un experimento con Skippers-CCD con los programas Geant4, MCNP y PHITS
          \item<+-> Se comienza trabajando en el rango de energías más cómodo (keV - MeV) donde estos programas suelen funcionar muy bien
        \end{itemize}
\end{frame}

\begin{frame}{Rangos de trabajo de los programas}
    \begin{center}
        Energías y partículas simulados por MCNP6
        \includegraphics[height=5cm]{energy_range_mcnp.png}
    \end{center}
    \begin{itemize}
        \item Tanto PHITS como MCNP6 simulan fotones y electrones hasta 1keV (simulación actual). 
        \item Con MCNP6 se puede bajar hasta 1eV (fotones) y 10eV (electrones).
        \item Geant4 puede utilizar al código PENELOPE para simular bajas energías.
    \end{itemize}
\end{frame}

\section{Análisis}

\begin{frame}{Fotones emitidos por la fuente de $^{252}$Cf}
    \begin{itemize}
        \item Por cada fisión espontánea se emiten: $\bar{\nu}_n = 3.7$ y $\bar{\nu}_p = 8.3$
        \item Espectro de fisión (Watt) para neutrones ($\bar E \approx 2MeV$)
        \item Se asume una fuente sin quemado
    \end{itemize}
    \begin{columns}[t]
        \begin{column}{0.5\textwidth}
            \begin{center}
                \vspace{-3em}
            \includegraphics[width=8cm]{espectros_fuenteCf_LLNL.png}\\
            \end{center}
        \end{column}
        \begin{column}{0.5\textwidth}
            Algunas preguntas:
            \begin{itemize}
                \item ¿Cuánto influyen estos fotones en el detector?
                \item Por otro lado, los neutrones generan fotones al interactuar con el resto de los materiales
                \item ¿Qué contribución de fotones es más importante?
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}


\begin{frame}{Algunas respuestas}
